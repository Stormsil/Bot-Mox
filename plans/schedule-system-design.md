function validateSessions(sessions: ScheduleSession[]): string[] {
  const errors: string[] = [];
  
  // Sort by start time
  const sorted = [...sessions].sort((a, b) => 
    a.start.localeCompare(b.start)
  );
  
  for (let i = 0; i < sorted.length; i++) {
    const session = sorted[i];
    
    // Check format
    if (!/^([0-1]?[0-9]|2[0-3]):[0-5][0-9]$/.test(session.start)) {
      errors.push(`Invalid start time format: ${session.start}`);
    }
    
    // Check duration (minimum 15 minutes)
    const startMin = timeToMinutes(session.start);
    const endMin = timeToMinutes(session.end);
    if (endMin - startMin < 15) {
      errors.push(`Session too short: ${session.start}-${session.end}`);
    }
    
    // Check overlaps
    if (i > 0) {
      const prevEnd = timeToMinutes(sorted[i-1].end);
      if (startMin < prevEnd) {
        errors.push(`Session overlap: ${sorted[i-1].start}-${sorted[i-1].end} and ${session.start}-${session.end}`);
      }
    }
  }
  
  return errors;
}
